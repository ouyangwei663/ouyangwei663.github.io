    # 实现react + ant-design动态更改主题颜色

    # 参考多方资料，具体实现流程

    实现思路：主要依靠ant-design的less变量，通过去修改less变量，实现样式的更新。less的版本信息如下

    "less": "^4.1.1",
"less-loader": "^5.0.0",

    # 1、使用 npm run eject 开启webpack的配置文件（如果已经开启请忽视）

    # 2、安装 antd-theme-generator 依赖

    npm i antd-theme-generator --save

    # 3、在src/assets文件夹下新增theme（自定义文件夹），在theme里新建color.less文件，在这里文件里配置默认的主题颜色

    @primary-color: rgb(24, 144, 255); // 全局主色
@link-color: rgb(24, 144, 255); // 链接色
@success-color: rgb(82, 196, 26); // 成功色
@warning-color: rgb(250, 173, 20); // 警告色
@error-color: rgb(245, 34, 45); // 错误色
@layout-text: rgb(241, 240, 240); // 布局字体色
@layout-background: rgba(0, 0, 0, 0.85); // 布局背景色
@heading-color: rgba(0, 0, 0, 0.85); // 标题色
@text-color: rgba(0, 0, 0, 0.65); // 主文本色
@text-color-secondary: rgba(0, 0, 0, 0.45); // 次文本色
@disabled-color: rgba(0, 0, 0, 0.25); // 失效色
@border-color-base: #d9d9d9; // 边框色

    # 4、在项目根目录下创建color.js 文件,代码如下

    const { generateTheme } = require("antd-theme-generator");
const path = require("path");

const options = {
  antDir: path.join(__dirname, "./node_modules/antd"),
  stylesDir: path.join(__dirname, "./src"), // 将处理扩展名为.less的所有文件
  varFile: path.join(__dirname, "./src/assets/theme/color.less"), 
  // 需要更改的属性
  themeVariables: [
    "@primary-color",
    "@link-color",
    "@success-color",
    "@warning-color",
    "@error-color",
    "@layout-text",
    "@layout-background",
    "@heading-color",
    "@text-color",
    "@text-color-secondary",
    "@disabled-color",
    "@border-color-base",
  ],
  outputFilePath: path.join(__dirname, "./public/color.less"),
};

generateTheme(options)
  .then((less) => {
    console.log("Theme generated successfully");
  })
  .catch((error) => {
    console.log("Error", error);
  });

    # 5、在index.html 文件中的body添加如下代码，在window下添加less，方便在组件中配合使用window.less.modifyVars去更改颜色

    <body>

  <script>
    window.less = { async: false, env: "production" };
  </script>
  <script type="text/javascript" src="https://cdn.bootcss.com/less.js/2.7.3/less.min.js"></script>

    # 7 到这一步基本上已经大功告成了。只需要在需要的组件进行更改就可以了，代码如下

    // 如果只需要修改部分，可根据自己的需要去修改
 export function ModifyThemeColor(color: string): void {
  window.less.modifyVars({
    "@primary-color": color, // 全局主色
    "@link-color": color, // 链接色
    "@success-color": color, // 成功色
    "@warning-color": color, // 警告色
    "@error-color": color, // 错误色
    "@layout-text": color, // 布局字体色
    "@layout-background": color, // 布局背景色
    "@heading-color": color, // 标题色
    "@text-color": color, // 主文本色
    "@text-color-secondary": color, // 次文本色
    "@disabled-color": color, // 失效色
    "@border-color-base": color, // 边框色
  });
  // 组件应用
  const colorMenu = (
  <Menu>
    {themeColor.map(item => {
      return (
        <Menu.Item key={item.color} onClick={() => ModifyThemeColor(item.color)}>
          {item.colorName}
        </Menu.Item>
      )
    })}
  </Menu>
)
}

    # 测试发现的问题

    # 1、在项目启动后再去修改src/assets/theme/color.less里的变量热更新会生效，需要重新启动项目才能更改。以及要更改的主题颜色需要先给默认值。否则不能在组件里动态更新主题

    # 2、页面刷新不能保存刚才设置好的主题颜色，会重新回归至默认主题.

    # 解决思路，将less变量储存在本地储存，在下次刷新页面的时候先读取本地储存进行更新主题颜色，在app.tsx添加代码如下

      const themeColor = localStorage.getItem("themeColor");
  themeColor && ModifyThemeColor(themeColor);
